CREATE DATABASE IF NOT EXISTS bettingshop;
USE bettingshop;
DROP TABLE IF EXISTS horsebet;
CREATE TABLE horsebet (
    id INTEGER AUTO_INCREMENT,
    horsename VARCHAR(30),
    place VARCHAR(30),
    time TIME,
    dateofrace DATE,
    stake INT,
    PRIMARY KEY (id)
);
